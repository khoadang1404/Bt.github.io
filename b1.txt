import java.util.Scanner;
public class Bai1_TongPS {
    public static void main(String[] args) {
        Scanner dt = new Scanner(System.in);
        System.out.println("nhap n: ");
        int so = dt.nextInt();
        TONGPS(so);
    }
    public static void TONGPS(int so) {
        float tong =0;
        for(int i=1; i<=so; i++)
            tong = tong + 1/(float)i;
        System.out.printf("tong phan so = %.2f", tong);
    }
}
